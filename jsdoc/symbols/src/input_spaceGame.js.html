 90 /**
 91  * Creates a sound effect which can be played multiple times concurrently.
 92  * @param {string} filename Name of sound file.
 93  * @constructor
 94  */
 95 function SoundEffect(filename) {
 96     /**
 97      * Filename of sound effect.
 98      * @type {string}
 99      */
100     this.filename = filename;
101 
102     /**
103      * Array of copies of audio to use to play sound effect multiple times at once.
104      * @type {Array}
105      */
106     this.sounds = [];
107 
108     /**
109      * True if sound is enabled.
110      * @type {boolean}
111      */
112     this.soundOn = true;
113 
114     for(var i = 0; i < 10; i++)
115         this.sounds.push(new Audio(this.filename));
116 
117     /**
118      * Index corresponding to Audio element in sounds array that should
119      * be played next.
120      * @type {number}
121      */
122     this.indexToPlay = 0;
123 }
124 
125 /**
126  * Plays an instance of this sound effect.
127  */
128 SoundEffect.prototype.play = function () {
129     if (this.soundOn) {
130         this.sounds[this.indexToPlay].play();
131         this.indexToPlay++;
132         if (this.indexToPlay > this.sounds.length - 1) {
133             this.indexToPlay = 0;
134         }
135     }
136 };
