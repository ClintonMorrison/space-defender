 13 /**
 14  * Used to cause an object to inherit from another object.
 15  * @param ParentClassOrObject The object or parent to inherit from.
 16  * @returns {Function}
 17  * @author Gavin Kistner (code taken from http://phrogz.net/JS/classes/OOPinJS2.html )
 18  */
 19 Function.prototype.inheritsFrom = function (ParentClassOrObject) {
 20     if (ParentClassOrObject.constructor === Function) {
 21         //Normal Inheritance
 22         this.prototype = new ParentClassOrObject();
 23         this.prototype.constructor = this;
 24         this.prototype.parent = ParentClassOrObject.prototype;
 25     } else {
 26         //Pure Virtual Inheritance
 27         this.prototype = ParentClassOrObject;
 28         this.prototype.constructor = this;
 29         this.prototype.parent = ParentClassOrObject;
 30     }
 31     return this;
 32 };

 38 /**

 59 /**

 62  * @return {Vector} A vector connecting vector v to this vector.

 64 Vector.prototype.vectorFrom = function (v) {
 65     return new Vector(this.x - v.x, this.y - v.y);

 68 /**
 69  * Returns a vector with length 1 parallel to this vector.
 70  * @return {Vector} A normalized copy of this vector.
 71  */
 72 Vector.prototype.getNormalized = function () {
 73     var magnitude = Math.sqrt(this.x * this.x + this.y * this.y);
 74 
 75     if (magnitude == 0.0)
 76         return new Vector(0, 0);
 77 
 78     return new Vector(this.x / magnitude, this.y / magnitude);
 79 };

126  * @param {Vector} vec The vector to get the distance to.
127  * @return {number} The distance to the vector v.
128  */
129 Vector.prototype.distanceFrom = function(vec) {
130     var s = new Vector(this.x - vec.x, this.y-vec.y);

134 /**
135  * Returns the angle this vector makes in the unit circle.
136  * @return {number} The angle of the vector.
137  */
138 Vector.prototype.getAngle = function () {
139     var theta = Math.atan(Math.abs(this.y/this.x)) * 180 / Math.PI;
140 
141     if (this.x <= 0 && this.y >= 0) theta = 180 - theta;
142     else if (this.x <= 0 && this.y <= 0) theta = 180 + theta;
143     else if(this.x >= 0 && this.y <= 0) theta = 360 - theta;
144 
145     return theta;
146 };

408 /**
409  * Creates a star object. Stars are small white
410  * rectangles drawn to screen. To simulate depth
411  * stars that are larger are moved faster than those
412  * that appear smaller.
413  * @param {Vector} position Position of star.
414  * @param {Vector} velocity Velocity of star.
415  * @param {Vector} windowSize Size of game window.
416  * @constructor
417  */
418 function Star(position, velocity, windowSize) {
419 
420     Sprite.call(this, position, velocity);
421     this.maxLength = 4;
422     this.length = Math.random() * this.maxLength;
423     this.outOfBounds = false;
424     this.color = "white";
425     this.drawingPosition = new Vector(0, 0);
426     this.shake = false;
427 
428     this.velocity = this.velocity.getScaled(this.length / this.maxLength);
429 }
430 
431 /**
432  * Controls if stars should shake randomly about
433  * their current position or not.
434  * @param {boolean} shake True if stars should shake.
435  */
436 Star.prototype.setShake = function(shake) {
437     this.shake = shake;
438 };
439 
440 /**
441  * Draws star to canvas.
442  * @param {CanvasRenderingContext2D} ctx Drawing context to draw star to.
443  */
444 Star.prototype.draw = function (ctx) {
445     this.drawingPosition = new Vector(this.position.x - this.offset.x * this.length / this.maxLength,
446         this.position.y - this.offset.y * this.length / this.maxLength);
447     ctx.fillStyle = this.color;
448 
449     var screenX = this.drawingPosition.x;
450     var screenY = this.drawingPosition.y;
451 
452     while (screenX < 0) screenX += windowSize.x;
453     while (screenX > windowSize.x) screenX -= windowSize.x;
454 
455     while (screenY < 0) screenY += windowSize.y;
456     while (screenY > windowSize.y) screenY -= windowSize.y;
457 
458     if (this.shake) {
459         screenX += Math.random() * 5;
460         screenY += Math.random() * 5;
461     }
462 
463     ctx.fillRect(screenX, screenY, this.length, this.length);
464 };
465 
466 /**
467  * Moves and updates star. Flags star as being
468  * out of bounds if star is out of bounds.
469  * @param {number} time Time (ms) since last update.
470  */
471 Star.prototype.update = function (time) {
472 
473     Sprite.prototype.update.call(this, time);
474 
475     if (this.position.x  < 0 || this.position.x > windowSize.x
476         || this.position.y < 0 || this.position.y > windowSize.y) {
477         this.outOfBounds = true;
478     }
479 };

645 /**
646  * Draws button.
647  * @param {CanvasRenderingContext2D} ctx Context to draw button to.
648  */
649 Button.prototype.draw = function (ctx) {
650     //Draw background
651     ctx.beginPath();
652     ctx.fillStyle = this.backgroundColor;
653 
654     ctx.arc(this.relativePosition.x * windowSize.x, this.relativePosition.y * windowSize.y,
655         this.radius, 0, 2 * Math.PI);
656 
657     ctx.closePath();
658     ctx.fill();
659 
660     //Draw foreground
661     ctx.beginPath();
662     if (this.highlighted) ctx.strokeStyle = this.highlightColor;
663     else ctx.strokeStyle = this.foregroundColor;
664     ctx.lineWidth = this.edgeThickness;
665 
666     ctx.arc(this.relativePosition.x * windowSize.x, this.relativePosition.y*windowSize.y,
667         this.radius, 0, 2 * Math.PI);
668 
669     ctx.closePath();
670     ctx.stroke();
671 
672     //Draw text
673     ctx.font = "bold " + this.fontSize + "px sans-serif";
674     if (this.highlighted) ctx.fillStyle = this.highlightColor;
675     else ctx.fillStyle = this.foregroundColor;
676     var textMeasure = ctx.measureText(this.text);
677     var singleLine = true;
678 
679     if(textMeasure.width > this.radius*2) {
680         singleLine = false;
681     }
682 
683     if (singleLine) {
684         var textBox = createCenteredRectangle(this.relativePosition.x * windowSize.x,
685             this.relativePosition.y * windowSize.y + 1.5 * this.fontSize / 4, textMeasure.width, 0);
686         ctx.fillText(this.text, textBox.x, textBox.y);
687     }
688     if(!singleLine) {
689 
690         var words = this.text.split(" ");
691         var width1 = ctx.measureText(words[0]).width;
692         var width2 = ctx.measureText(words[1]).width;
693 
694         var textBox1 = createCenteredRectangle(this.relativePosition.x * windowSize.x,
695             this.relativePosition.y * windowSize.y + 1.5 * this.fontSize / 4 - 6, width1, 0);
696 
697         var textBox2 = createCenteredRectangle(this.relativePosition.x * windowSize.x,
698             this.relativePosition.y * windowSize.y + 1.5 * this.fontSize / 2 + 6, width2, 0);
699 
700         if (words.length == 2) {
701             ctx.fillText(words[0], textBox1.x, textBox1.y);
702             ctx.fillText(words[1], textBox2.x, textBox2.y);
703         }
704     }
705 };
